# Python slim + tesseract OCR (spa/cat) para imágenes
FROM python:3.11-slim

# Evitar prompts y reducir tamaño
ENV DEBIAN_FRONTEND=noninteractive \
    PYTHONUNBUFFERED=1

# Paquetes del sistema necesarios (tesseract + idiomas + build deps mínimos)
RUN apt-get update && apt-get install -y --no-install-recommends \
    tesseract-ocr \
    tesseract-ocr-spa \
    tesseract-ocr-cat \
    libjpeg62-turbo \
    zlib1g \
    && rm -rf /var/lib/apt/lists/*

# Directorio de la app
WORKDIR /app

# Requisitos
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copia el resto del proyecto
COPY . .

# Puerto interno (Render ignora EXPOSE pero es informativo)
EXPOSE 10000

# Arranque del backend (ajusta si tu factory difiere)
CMD ["gunicorn", "app:create_app()", "--bind", "0.0.0.0:10000", "--workers", "2"]
